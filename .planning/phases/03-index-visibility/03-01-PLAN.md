Target contract this plan should establish for downstream plans:

```ts
export type IndexStage = "crawl" | "process" | "chunk/embed" | "persist";

export type SkipReasonBucket =
  | "large_file"
  | "binary_file"
  | "ignored_json"
  | "no_indexable_chunks"
  | "processing_error";

export interface ScanStats {
  totalFiles: number;
  added: number;
  modified: number;
  unchanged: number;
  deleted: number;
  skipped: number;
  errors: number;
  skippedByReason: Partial<Record<SkipReasonBucket, number>>;
  visibility: {
    candidateFiles: number;
    processedFiles: number;
    embeddingFiles: number;
    selfHealFiles: number;
    deletedPaths: number;
  };
  vectorIndex?: {
    indexed: number;
    deleted: number;
    errors: number;
  };
}

export class ScanStageError extends Error {
  readonly stage: IndexStage;
  readonly partialStats?: ScanStats;
}
```
