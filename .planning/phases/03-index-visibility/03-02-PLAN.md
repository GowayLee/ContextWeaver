onProgress: (current, total, message) => {
  if (total === undefined) return;
  const percent = Math.floor((current / total) * 100);
  if (percent >= lastLoggedPercent + 30 && percent < 100) {
    logLine(`索引进度: ${percent}% - ${message || ""}`);
  }
};
